function showPage(page) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById(`page-${page}`).classList.add('active');
  event.target.classList.add('active');

  if (page === 'flashcards') initFlashcards();
}

// Render disease grid
function renderGrid() {
  const grid = document.getElementById('diseaseGrid');
  grid.innerHTML = '';
  diseases.forEach(d => {
    const card = document.createElement('div');
    card.className = 'disease-card';
    card.style.setProperty('--card-color', d.color);
    card.style.borderLeftColor = d.color;
    card.style.borderLeftWidth = '3px';
    card.innerHTML = `
      <div class="card-tag">${d.category}</div>
      <div class="card-title">${d.title}</div>
      <div class="card-preview">${d.preview}</div>
      <div class="card-arrow">→</div>
    `;
    card.onclick = () => openModal(d);
    grid.appendChild(card);
  });
}

function openModal(d) {
  document.getElementById('modalTag').textContent = d.category;
  document.getElementById('modalTag').style.color = d.color;
  document.getElementById('modalTitle').textContent = d.title;

  const body = document.getElementById('modalBody');
  body.innerHTML = `
    <div class="section-block">
      <h3>📊 Epidemiologia</h3>
      <p>${d.epidemio}</p>
    </div>
    <div class="section-block">
      <h3>⚙️ Fisiopatologia → Clínica</h3>
      <p>${d.fisio}</p>
      <p style="margin-top:0.6rem">${d.clinica}</p>
    </div>
    <div class="section-block">
      <h3>🔬 Exames</h3>
      <ul class="pill-list">${d.exames.map(e => `<li class="pill">${e}</li>`).join('')}</ul>
    </div>
    <div class="section-block">
      <h3>💊 Tratamento</h3>
      <p>${d.tratamento}</p>
    </div>
    <div class="alert-box">⚠️ <strong>Alerta / Encaminhamento:</strong> ${d.alerta}</div>
  `;

  document.getElementById('modalOverlay').classList.add('open');
  document.body.style.overflow = 'hidden';
}

function closeModal(e) {
  if (e.target === document.getElementById('modalOverlay')) closeModalBtn();
}

function closeModalBtn() {
  document.getElementById('modalOverlay').classList.remove('open');
  document.body.style.overflow = '';
}

// FLASHCARDS
function initFlashcards() {
  // Populate filter
  const filter = document.getElementById('fcFilter');
  const cats = [...new Set(flashcards.map(f => f.cat))];
  filter.innerHTML = '<option value="all">Todas as categorias</option>';
  cats.forEach(c => {
    const opt = document.createElement('option');
    opt.value = c; opt.textContent = c;
    filter.appendChild(opt);
  });
  updateCard();
  updateStats();
}

function filterCards() {
  const val = document.getElementById('fcFilter').value;
  currentCards = val === 'all' ? [...flashcards] : flashcards.filter(f => f.cat === val);
  currentCardIndex = 0;
  isFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  document.getElementById('rateButtons').style.display = 'none';
  updateCard();
}

function updateCard() {
  if (currentCards.length === 0) return;
  const card = currentCards[currentCardIndex];
  document.getElementById('fcCategory').textContent = card.cat;
  document.getElementById('fcQuestion').textContent = card.q;
  document.getElementById('fcAnswer').innerHTML = card.a;
  document.getElementById('fcProgress').textContent = `${currentCardIndex + 1} / ${currentCards.length}`;
  document.getElementById('stat-total').textContent = currentCards.length;
}

function flipCard() {
  isFlipped = !isFlipped;
  document.getElementById('flashcard').classList.toggle('flipped', isFlipped);
  document.getElementById('rateButtons').style.display = isFlipped ? 'flex' : 'none';
}

function nextCard() {
  if (currentCardIndex < currentCards.length - 1) currentCardIndex++;
  else currentCardIndex = 0;
  isFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  document.getElementById('rateButtons').style.display = 'none';
  updateCard();
}

function prevCard() {
  if (currentCardIndex > 0) currentCardIndex--;
  else currentCardIndex = currentCards.length - 1;
  isFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  document.getElementById('rateButtons').style.display = 'none';
  updateCard();
}

function rateCard(rating) {
  if (rating === 'good') stats.good++;
  else if (rating === 'wrong') stats.wrong++;
  updateStats();
  nextCard();
}

function updateStats() {
  document.getElementById('stat-good').textContent = stats.good;
  document.getElementById('stat-wrong').textContent = stats.wrong;
}

function shuffleCards() {
  currentCards = currentCards.sort(() => Math.random() - 0.5);
  currentCardIndex = 0;
  isFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  document.getElementById('rateButtons').style.display = 'none';
  updateCard();
}

function resetStats() {
  stats = { good: 0, wrong: 0 };
  currentCardIndex = 0;
  isFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  document.getElementById('rateButtons').style.display = 'none';
  updateStats();
  updateCard();
}

let farmaFilter = 'all';

function setFarmaFilter(cat, btn) {
  farmaFilter = cat;
  document.querySelectorAll('.farma-filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  renderFarma();
}

function renderFarma() {
  const search = document.getElementById('farmaSearch').value.toLowerCase();
  const tbody = document.getElementById('farmaTableBody');

  let data = farmaData;
  if (farmaFilter !== 'all') data = data.filter(d => d.cat === farmaFilter);
  if (search) data = data.filter(d =>
    d.drug.toLowerCase().includes(search) ||
    d.classe.toLowerCase().includes(search) ||
    d.indicacao.toLowerCase().includes(search) ||
    d.obs.toLowerCase().includes(search)
  );

  if (data.length === 0) {
    tbody.innerHTML = `<tr><td colspan="6" class="no-results">Nenhum medicamento encontrado.</td></tr>`;
    return;
  }

  const tagMap = { CV:'tag-cv', Meta:'tag-meta', Resp:'tag-resp', Psiq:'tag-psiq', Infec:'tag-infec', End:'tag-end', Orto:'tag-orto' };
  const catLabel = { CV:'Cardiovascular', Meta:'Metabólico', Resp:'Respiratório', Psiq:'Psiquiatria', Infec:'Infecciologia', End:'Endocrinologia', Orto:'Ortop./Musc.' };

  tbody.innerHTML = data.map(d => `
    <tr>
      <td>
        <div class="farma-drug">${d.drug}</div>
        <div class="farma-class" style="color:var(--text-muted); margin-top:0.2rem">${d.classe}</div>
      </td>
      <td><span class="farma-tag ${tagMap[d.cat]}">${catLabel[d.cat]}</span></td>
      <td style="color:var(--text-dim)">${d.indicacao}</td>
      <td><span class="farma-dose">${d.dose}</span></td>
      <td style="color:var(--text-dim)">
        ${d.ci}
        ${d.alerta === 'ci' ? `<div class="farma-alert">⚠ Atenção às contraindicações</div>` : ''}
      </td>
      <td>
        ${d.obs}
        ${d.alerta === 'tip' ? `<div class="farma-tip">💡 Dica clínica</div>` : ''}
      </td>
    </tr>
  `).join('');
}

let rastreioView = 'cancer';

function setRastreioView(view, btn) {
  rastreioView = view;
  document.querySelectorAll('.rastreio-filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  renderRastreio();
}

function renderRastreio() {
  const container = document.getElementById('rastreio-content');
  const data = rastreioData[rastreioView];
  if (!data) return;

  if (rastreioView === 'faixa') {
    container.innerHTML = `
      <div class="faixa-etaria-grid">
        ${data.faixas.map(f => `
          <div class="faixa-card">
            <div class="faixa-header" style="background: linear-gradient(135deg, ${f.cor}15, ${f.cor}05); border-bottom: 2px solid ${f.cor}40">
              <div class="faixa-icon">${f.icon}</div>
              <div>
                <div class="faixa-title" style="color:${f.cor}">${f.faixa}</div>
                <div class="faixa-subtitle">${f.subtitulo}</div>
              </div>
            </div>
            <div class="faixa-body">
              ${f.exames.map(e => `
                <div class="faixa-item">
                  <div class="faixa-dot" style="background:${f.cor}"></div>
                  <div class="faixa-exam">${e.exam}</div>
                  <div class="faixa-freq">${e.freq}</div>
                </div>
              `).join('')}
            </div>
          </div>
        `).join('')}
      </div>
    `;
    return;
  }

  container.innerHTML = `
    <div class="rastreio-section">
      <div class="rastreio-section-header">
        <span class="rastreio-section-icon">${data.icon}</span>
        <span class="rastreio-section-title">${data.title}</span>
      </div>
      <div class="rastreio-cards">
        ${data.items.map(item => `
          <div class="rastreio-card" style="--rc-color: ${item.color}">
            <div class="rc-header">
              <div class="rc-title">${item.title}</div>
              <div class="rc-population" style="background:${item.color}15; border-color:${item.color}; color:${item.color}">${item.populacao}</div>
            </div>
            <div class="rc-row">
              <span class="rc-label">Método</span>
              <span class="rc-value">${item.metodo}</span>
            </div>
            <div class="rc-row">
              <span class="rc-label">Intervalo</span>
              <span class="rc-value">${item.intervalo}</span>
            </div>
            <div class="rc-row">
              <span class="rc-label">Início</span>
              <span class="rc-value">${item.inicio}</span>
            </div>
            <div class="rc-row">
              <span class="rc-label">Fim</span>
              <span class="rc-value">${item.fim}</span>
            </div>
            ${item.note ? `<div class="rc-note">💡 ${item.note}</div>` : ''}
            ${item.alerta ? `<div class="rc-alert">⚠️ ${item.alerta}</div>` : ''}
          </div>
        `).join('')}
      </div>
    </div>
  `;
}
